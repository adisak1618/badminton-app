    **2. Create apps/web/components/liff/liff-provider.tsx** — Client component per RESEARCH.md Pattern 1 + Pattern 2:

    ```typescript
    "use client";
    import { createContext, useContext, useEffect, useState, useCallback, type ReactNode } from "react";
    import type { Liff } from "@line/liff";

    interface LiffContextValue {
      liff: Liff | null;
      liffError: string | null;
      isReady: boolean;
      isLoggedIn: boolean;
    }

    const LiffContext = createContext<LiffContextValue>({
      liff: null,
      liffError: null,
      isReady: false,
      isLoggedIn: false,
    });

    export function LiffProvider({ liffId, children }: { liffId: string; children: ReactNode }) {
      const [liff, setLiff] = useState<Liff | null>(null);
      const [liffError, setLiffError] = useState<string | null>(null);
      const [isReady, setIsReady] = useState(false);
      const [isLoggedIn, setIsLoggedIn] = useState(false);

      useEffect(() => {
        import("@line/liff")
          .then((mod) => mod.default)
          .then((liffInstance) =>
            liffInstance.init({ liffId }).then(() => {
              setLiff(liffInstance);

              // After LIFF init, get ID token and authenticate with our server
              const idToken = liffInstance.getIDToken();
              if (idToken) {
                fetch("/api/auth/liff", {
                  method: "POST",
                  headers: { "Content-Type": "application/json" },
                  body: JSON.stringify({ idToken }),
                })
                  .then((res) => res.json())
                  .then((data) => {
                    setIsLoggedIn(true);
                    setIsReady(true);
                  })
                  .catch((err) => {
                    setLiffError("Authentication failed");
                    setIsReady(true);
                  });
              } else {
                setLiffError("Could not get ID token. Please ensure this page is opened inside LINE.");
                setIsReady(true);
              }
            })
          )
          .catch((err) => {
            setLiffError(err.toString());
            setIsReady(true);
          });
      }, [liffId]);

      return (
        <LiffContext.Provider value={{ liff, liffError, isReady, isLoggedIn }}>
          {children}
        </LiffContext.Provider>
      );
    }

    export function useLiff() {
      return useContext(LiffContext);
    }
    ```

    Key design decisions:
    - Dynamic `import("@line/liff")` inside useEffect to avoid SSR `window is not defined` crash (Pitfall 1 from RESEARCH.md)
    - Calls `liff.getIDToken()` after init and POSTs to `/api/auth/liff` (our own Next.js route, NOT through proxy — same origin) per D-01
    - Does NOT use `/api/proxy/liff/auth` — the auth route is a Next.js route handler at `/api/auth/liff`, not an Elysia route
    - `isReady` flag indicates both LIFF init AND server auth have completed (or errored)
    - `liffId` prop comes from `process.env.NEXT_PUBLIC_LIFF_ID` passed from the layout
    - Per D-03: no browser fallback — if liff.getIDToken() returns null, show error

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| LIFF client -> /api/auth/liff | Untrusted ID token crosses from client to server |
| /api/auth/liff -> LINE /verify | Server sends token to LINE for verification |
| Client -> /api/liff/profile | Profile data from client crosses to API |
| authMiddleware -> session cookie | Sealed cookie is the trust boundary for API routes |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Spoofing | /api/auth/liff | mitigate | Never trust client-supplied lineUserId — extract only from LINE-verified ID token `profile.sub` |
| T-03-02 | Tampering | /api/auth/liff | mitigate | Verify ID token server-side via LINE `/oauth2/v2.1/verify` with `client_id` check; reject if `!verifyRes.ok` |
| T-03-03 | Spoofing | /api/liff/profile | mitigate | All profile routes require authMiddleware — lineUserId extracted from sealed session, not request body |
| T-03-04 | Tampering | /api/liff/profile POST/PUT | mitigate | Validate all input with Elysia `t.Object` schema — `t.String({ minLength: 1 })`, `t.Union([t.Literal(...)])`, `t.Integer({ minimum: 0 })` |
| T-03-05 | Information Disclosure | /api/liff/profile GET | mitigate | Return only displayName, skillLevel, yearsPlaying — no lineUserId or internal IDs leaked beyond memberId |
| T-03-06 | Elevation of Privilege | /api/liff/profile PUT | mitigate | Update only the member record matching `session.lineUserId` — user cannot modify another user's profile |
| T-03-07 | Replay | /api/auth/liff | accept | LINE /verify checks token expiry (`exp` claim); expired tokens are rejected automatically. Short-lived ID tokens (~5 min) limit replay window. |
</threat_model>
