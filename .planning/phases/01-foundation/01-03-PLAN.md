```typescript
import { describe, it, expect, beforeAll, afterAll } from "bun:test";
import { db, clubs, events } from "@repo/db";
import { eq } from "drizzle-orm";

describe("cross-tenant isolation", () => {
  let clubAId: string;
  let clubBId: string;

  beforeAll(async () => {
    // Seed two clubs
    const [clubA] = await db.insert(clubs).values({
      name: "Test Club A",
    }).returning();
    const [clubB] = await db.insert(clubs).values({
      name: "Test Club B",
    }).returning();
    clubAId = clubA!.id;
    clubBId = clubB!.id;

    // Create an event for Club A only
    await db.insert(events).values({
      clubId: clubAId,
      title: "Club A Event",
      eventDate: new Date("2026-05-01T18:00:00+07:00"),
    });
  });

  afterAll(async () => {
    // Cleanup: delete test events then clubs
    await db.delete(events).where(eq(events.clubId, clubAId));
    await db.delete(events).where(eq(events.clubId, clubBId));
    await db.delete(clubs).where(eq(clubs.id, clubAId));
    await db.delete(clubs).where(eq(clubs.id, clubBId));
  });

  it("club A events are not visible when querying with club B id", async () => {
    const clubBEvents = await db
      .select()
      .from(events)
      .where(eq(events.clubId, clubBId));

    expect(clubBEvents).toHaveLength(0);
  });

  it("club A events are visible when querying with club A id", async () => {
    const clubAEvents = await db
      .select()
      .from(events)
      .where(eq(events.clubId, clubAId));

    expect(clubAEvents).toHaveLength(1);
    expect(clubAEvents[0]!.title).toBe("Club A Event");
  });
});
```

```typescript
import { describe, it, expect, afterAll } from "bun:test";
import { db, idempotencyKeys } from "@repo/db";

describe("webhook idempotency", () => {
  const testEventId = "test-idem-" + Date.now();

  afterAll(async () => {
    await db.delete(idempotencyKeys).where(
      eq(idempotencyKeys.webhookEventId, testEventId)
    );
  });

  it("first insert returns the inserted row", async () => {
    const result = await db
      .insert(idempotencyKeys)
      .values({ webhookEventId: testEventId })
      .onConflictDoNothing()
      .returning();

    expect(result).toHaveLength(1);
    expect(result[0]!.webhookEventId).toBe(testEventId);
  });

  it("duplicate insert returns empty array (conflict)", async () => {
    const result = await db
      .insert(idempotencyKeys)
      .values({ webhookEventId: testEventId })
      .onConflictDoNothing()
      .returning();

    expect(result).toHaveLength(0);
  });
});
```

4. Create `apps/api/src/__tests__/webhook.test.ts`:
```typescript
import { describe, it, expect } from "bun:test";
import { createHmac } from "crypto";
import app from "../index";

// Generate a valid LINE signature for testing
function generateSignature(body: string, secret: string): string {
  return createHmac("sha256", secret).update(body).digest("base64");
}

describe("POST /webhook/line", () => {
  const channelSecret = process.env.LINE_CHANNEL_SECRET!;

  it("returns 401 for invalid signature", async () => {
    const body = JSON.stringify({ events: [] });
    const response = await app.handle(
      new Request("http://localhost/webhook/line", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "x-line-signature": "invalid-signature",
        },
        body,
      })
    );

    expect(response.status).toBe(401);
  });

  it("returns 200 for valid signature", async () => {
    const body = JSON.stringify({ events: [], destination: "test" });
    const signature = generateSignature(body, channelSecret);

    const response = await app.handle(
      new Request("http://localhost/webhook/line", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "x-line-signature": signature,
        },
        body,
      })
    );

    expect(response.status).toBe(200);
  });

  it("returns 200 for GET /health", async () => {
    const response = await app.handle(
      new Request("http://localhost/health")
    );

    expect(response.status).toBe(200);
    const json = await response.json();
    expect(json).toEqual({ status: "ok" });
  });
});
```

NOTE: The `app.handle()` method is Elysia's built-in testing utility that processes a Request object without starting a server. This is the standard way to integration-test Elysia apps.
