4. Add the unlink handler function inside the component:
```typescript
const handleUnlink = async () => {
  setUnlinking(true);
  try {
    const res = await fetch(`/api/proxy/clubs/${params.id}/link`, {
      method: "DELETE",
    });
    if (!res.ok) throw new Error("Failed to unlink");
    toast.success("LINE Group unlinked");
    setDialogOpen(false);
    const updated = await fetch(`/api/proxy/clubs/${params.id}`);
    if (updated.ok) setClub(await updated.json());
  } catch {
    toast.error("Failed to unlink group. Please try again.");
    setDialogOpen(false);
  } finally {
    setUnlinking(false);
  }
};
```

5. In the JSX, after the `<ClubForm ... />` closing tag but still inside the `<div className="mx-auto max-w-lg">`, add:

    </p>
    <Dialog open={dialogOpen} onOpenChange={setDialogOpen}>
      <DialogTrigger asChild>
        <Button variant="destructive" className="mt-4">
          Unlink Group

      <DialogContent>
